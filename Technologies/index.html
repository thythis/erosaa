# Async function

# example

async componentDidMount() {
    try {
        const response = await 
            fetch(`https://api.github.com/repos/facebook/react/stargazers`);
        if (!response.ok) {
            throw Error(response.statusText);
        }
        const json = await response.json();
        this.setState({ data: json });
    } catch (error) {
        console.log(error);
    }
}
